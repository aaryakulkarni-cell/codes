-- Practical 1: SQL DDL and DML Demonstration
-- Using MySQL syntax for Ubuntu terminal

-- Create database and use it
CREATE DATABASE university_db;
USE university_db;

-- 1. Create Tables with constraints
CREATE TABLE Course (
  course_id INT PRIMARY KEY,
  course_name VARCHAR(50) NOT NULL
);

CREATE TABLE Student (
  student_id INT PRIMARY KEY,
  name VARCHAR(50) NOT NULL,
  age INT CHECK (age >= 18),
  course_id INT,
  FOREIGN KEY (course_id) REFERENCES Course(course_id)
);

-- 2. Insert data into Course table
INSERT INTO Course VALUES (1, 'DBMS');
INSERT INTO Course VALUES (2, 'Computer Networks');
INSERT INTO Course VALUES (3, 'Data Structures');
INSERT INTO Course VALUES (4, 'Operating Systems');

-- 3. Insert data into Student table
INSERT INTO Student VALUES (101, 'Arya', 20, 1);
INSERT INTO Student VALUES (102, 'Krushna', 21, 2);
INSERT INTO Student VALUES (103, 'Yash', 19, 1);
INSERT INTO Student VALUES (104, 'Priya', 22, 3);
INSERT INTO Student VALUES (105, 'Rahul', 20, 2);

-- 4. Create Index
CREATE INDEX idx_student_name ON Student(name);

-- QUERY 1: Basic SELECT with WHERE clause
SELECT * FROM Student WHERE name = 'Arya';

-- QUERY 2: SELECT with comparison operator
SELECT * FROM Student WHERE age > 20;

-- 5. Create Sequence (MySQL doesn't have sequences, using AUTO_INCREMENT alternative)
ALTER TABLE Student MODIFY student_id INT AUTO_INCREMENT;

-- Insert using auto increment
INSERT INTO Student (name, age, course_id) VALUES ('Rohit', 22, 2);

-- 6. Create View
CREATE VIEW student_view AS
SELECT name, age, course_id FROM Student WHERE age >= 20;

-- QUERY 3: SELECT from view
SELECT * FROM student_view;

-- QUERY 4: UPDATE statement
UPDATE Student SET age = 23 WHERE name = 'Rohit';

-- QUERY 5: DELETE statement
DELETE FROM Student WHERE name = 'Yash';

-- QUERY 6: JOIN between tables
SELECT s.name, s.age, c.course_name 
FROM Student s 
JOIN Course c ON s.course_id = c.course_id;

-- QUERY 7: Aggregate function with GROUP BY
SELECT c.course_name, COUNT(s.student_id) as student_count
FROM Course c 
LEFT JOIN Student s ON c.course_id = s.course_id
GROUP BY c.course_name;

-- QUERY 8: Using HAVING clause
SELECT c.course_name, COUNT(s.student_id) as student_count
FROM Course c 
LEFT JOIN Student s ON c.course_id = s.course_id
GROUP BY c.course_name
HAVING COUNT(s.student_id) > 0;

-- QUERY 9: Set operator (UNION)
SELECT name FROM Student WHERE course_id = 1
UNION
SELECT name FROM Student WHERE age > 21;

-- QUERY 10: Using string function
SELECT UPPER(name) as upper_name, age, LENGTH(name) as name_length FROM Student;

-- QUERY 11: Query with multiple conditions using AND/OR
SELECT * FROM Student WHERE age BETWEEN 20 AND 25 AND course_id IN (1, 2);

-- QUERY 12: Pattern matching with LIKE
SELECT * FROM Student WHERE name LIKE 'R%';

-- QUERY 13: Order by clause
SELECT * FROM Student ORDER BY age DESC;

-- QUERY 14: Using mathematical function
SELECT name, age, (age * 12) as age_in_months FROM Student;

-- Display all data
