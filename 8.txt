import java.sql.*;

public class SimplestDB {
    public static void main(String[] args) {
        try {
            // 1. CONNECT
            Connection conn = DriverManager.getConnection("jdbc:mysql://localhost:3306/student_db", "root", "");
            
            // 2. ADD
            Statement stmt = conn.createStatement();
            stmt.executeUpdate("INSERT INTO students (name, age) VALUES ('John', 20)");
            
            // 3. VIEW  
            ResultSet rs = stmt.executeQuery("SELECT * FROM students");
            while (rs.next()) {
                System.out.println("ID: " + rs.getInt("id") + " Name: " + rs.getString("name"));
            }
            
            // 4. EDIT
            stmt.executeUpdate("UPDATE students SET age=21 WHERE name='John'");
            
            // 5. DELETE
            stmt.executeUpdate("DELETE FROM students WHERE name='John'");
            
            conn.close();
            
        } catch (Exception e) {
            System.out.println("Error: " + e.getMessage());
        }
    }
}
